struct custom_hash{
    static uint64_t splitmix64(uint64_t x){
        x += 0x9e3779b97f4a7c15;
        x = (x^(x>>30))*0xbf58476d1ce4e5b9;
        x = (x^(x>>27))*0x94d049bb133111eb;
        return x^(x>>31);
    }

    template <typename T>
    size_t operator()(const T& x) const{
        static const uint64_t FIXED_RANDOM = chrono::steady_clock::now().time_since_epoch().count();
        return splitmix64(hash<T>{}(x) + FIXED_RANDOM);
    }

    template <typename T1, typename T2>
    size_t operator()(const pair<T1, T2>& p) const{
        uint64_t h1 = (*this)(p.first);
        uint64_t h2 = (*this)(p.second);
        return h1^(h2+0x9e3779b97f4a7c15+(h1<<6)+(h1>>2));
    }

    template <typename T>
    size_t operator()(const vector<T>& v) const{
        uint64_t hash_res = 0;
        for (const T& val : v){
            uint64_t h = (*this)(val);
            hash_res ^= h+0x9e3779b97f4a7c15+(hash_res << 6)+(hash_res >> 2);
        }
        return hash_res;
    }
};

ll bin(ll base, ll exp, ll mod){
    ll result = 1;
    base = base%mod;
    while(exp>0){
        if(exp&1){
            result = (result*base)%mod;
        }
        base = (base*base)%mod;
        exp >>= 1;
    }
    return result;
}

long long modInverse(long long a, long long mod) {
    return bin(a,mod-2,mod);
}

// Sieve of Eratosthenes.......................................................
vector<bool> isPrime;
void sieve(int n){
    isPrime.assign(n+1,true);
    isPrime[0]=isPrime[1]=false;
    for(int i=2;i*i<=n;i++){
        if(isPrime[i]){
            for(int j=i*i;j<=n;j+=i) isPrime[j]=false;
        }
    }
}

// Efficient prime check for large N.........................................
bool isPrime(ll n){
    if(n<=1) return false;
    if(n==2||n==3) return true;
    if(n%2==0||n%3==0) return false;
    for(int i=5; i*i<=n; i+=6){
        if(n%i==0||n%(i+2)==0) return false;
    }
    return true;
}

//DSU....................................................
vector<int> parent, ranks;
void makeSet(int n){
    parent.resize(n);
    ranks.resize(n,0);
    for(int i=0;i<n;i++) parent[i]=i;
}
int findSet(int v){
    if(v==parent[v]) return v;
    return parent[v]=findSet(parent[v]);
}
bool unionSets(int a,int b){
    a=findSet(a);
    b=findSet(b);
    if(a!=b){
        if(ranks[a]<ranks[b]) swap(a,b);
        parent[b]=a;
        if(ranks[a]==ranks[b]) ranks[a]++;
        return true;
    }
    return false;
}

class DoubleHash{
public:
    string s;
    ll mod1, mod2, omega1, omega2;
    vector<ll> prefix1, prefix2, power1, power2;

    // Predefined good bases and mods (use random bases for hacking resistance)
    DoubleHash(string str, ll base1 = 911382629, ll base2 = 357142857, 
               ll mod1 = 1e9+7, ll mod2 = 1e9+9) 
        : s(str), omega1(base1), omega2(base2), mod1(mod1), mod2(mod2){
        build();
    }

    void build(){
        int n = s.size();
        prefix1.assign(n+1,0);
        prefix2.assign(n+1,0);
        power1.assign(n+1,1);
        power2.assign(n+1,1);

        for (int i=0; i<n; i++){
            power1[i+1] = (power1[i]*omega1)%mod1;
            power2[i+1] = (power2[i]*omega2)%mod2;
            prefix1[i+1] = (prefix1[i]*omega1+s[i])%mod1;
            prefix2[i+1] = (prefix2[i]*omega2+s[i])%mod2;
        }
    }
    // Returns hash of s[l..r] (0-based, inclusive) as a pair
    pair<ll, ll> get_hash(int l, int r) {
        ll hash1 = (prefix1[r+1]-prefix1[l]*power1[r-l+1])%mod1;
        ll hash2 = (prefix2[r+1]-prefix2[l]*power2[r-l+1])%mod2;
        if (hash1<0) hash1 += mod1;
        if (hash2<0) hash2 += mod2;
        return {hash1, hash2};
    }
    // Computes hash of another string (t) using the same parameters
    pair<ll,ll> hash_string(string t){
        ll h1=0, h2=0;
        for(char c : t){
            h1 = (h1*omega1+c)%mod1;
            h2 = (h2*omega2+c)%mod2;
        }
        return {h1,h2};
    }
};

// KMP........................................

vector<int> buildPrefixTable(const string& pattern){
    int m = pattern.size();
    vector<int> prefix(m, 0);
    int j = 0;

    for(int i=1; i<m; ++i){
        if(pattern[i] == pattern[j]){
            prefix[i] = ++j;
        }else if(j>0){
            j = prefix[j-1];
            --i; // Retry this character
        }
    }
    return prefix;
}

// KMP algorithm
void KMP(const string& text, const string& pattern) {
    vector<int> prefix = buildPrefixTable(pattern);
    int n=text.size(), m=pattern.size();
    int i=0, j=0;

    while (i<n) {
        if(text[i] == pattern[j]){
            ++i; ++j;
        }
        if(j==m){
            cout<<"Pattern found at index "<<i-j<<endl;
            j = prefix[j-1];
        }else if(i<n && text[i] != pattern[j]){
            if(j>0){
                j = prefix[j-1];
            }else{
                ++i;
            }
        }
    }
}

//segment tree...............................................................

class segmentTree {
    vector<ll> t, lz;
    ll n;
    void push(ll node, ll s, ll e) {
        if(lz[node]==0) return;
        t[node]+=lz[node];
        if(s!=e) {
            lz[2*node+1]+=lz[node];
            lz[2*node+2]+=lz[node];
        }
        lz[node]=0;
    }
    void build(vector<ll>& arr, ll node, ll s, ll e) {
        if(s==e) {
            t[node]=arr[s];
            return;
        }
        ll mid=s+(e-s)/2;
        build(arr, 2*node+1, s, mid);
        build(arr, 2*node+2, mid+1, e);
        t[node]=min(t[2*node+1], t[2*node+2]);
    }
    void update(ll node, ll s, ll e, ll l, ll r, ll val) {
        push(node, s, e);
        if(s>r || e<l) return;
        if(s>=l && e<=r) {
            lz[node]+=val;
            push(node, s, e);
            return;
        }
        ll mid=s+(e-s)/2;
        update(2*node+1, s, mid, l, r, val);
        update(2*node+2, mid+1, e, l, r, val);
        t[node]=min(t[2*node+1], t[2*node+2]);
    }
    void pUpdate(ll node, ll s, ll e, ll idx, ll val) {
        push(node, s, e);
        if(s==e) {
            t[node]=val;
            return;
        }
        ll mid=s+(e-s)/2;
        if(idx<=mid) pUpdate(2*node+1, s, mid, idx, val);
        else pUpdate(2*node+2, mid+1, e, idx, val);
        push(2*node+1, s, mid);
        push(2*node+2, mid+1, e);
        t[node]=min(t[2*node+1], t[2*node+2]);
    }
    ll query(ll node, ll s, ll e, ll l, ll r) {
        push(node, s, e);
        if(s>r || e<l) return LLONG_MAX;
        if(s>=l && e<=r) return t[node];
        ll mid=s+(e-s)/2;
        return min(query(2*node+1, s, mid, l, r), query(2*node+2, mid+1, e, l, r));
    }
public:
    segmentTree(vector<ll>& arr) {
        n=arr.size();
        t.assign(4*n, 0);
        lz.assign(4*n, 0);
        build(arr, 0, 0, n-1);
    }
    void update(ll l, ll r, ll val) { update(0, 0, n-1, l, r, val); }
    void pointUpdate(ll idx, ll val) { pUpdate(0, 0, n-1, idx, val); }
    ll query(ll l, ll r) { return query(0, 0, n-1, l, r); }
};

//sparse table for idempotent operation O(1).................................

/**
 * Sparse Table for Idempotent Operations (Min, Max, GCD)
 * Construction: O(N log N)
 * Query: O(1)
**/

template <typename T>
class SparseTable {
    int n, K;
    vector<vector<T>> st;
    vector<int> logs;

    T combine(T a, T b) {
        return min(a, b); // Change this to max(a, b) or gcd(a, b) as needed
    }
public:
    SparseTable(const vector<T>& arr) {
        n = arr.size();
        K = __lg(n) + 1;
        st.assign(K, vector<T>(n));
        logs.assign(n + 1, 0);
        // Precompute logs for O(1) query
        for (int i = 2; i <= n; i++)
            logs[i] = logs[i / 2] + 1;

        for (int i = 0; i < n; i++)
            st[0][i] = arr[i];
        // Iterative construction
        for (int j = 1; j < K; j++) {
            for (int i = 0; i + (1 << j) <= n; i++) {
                st[j][i] = combine(st[j - 1][i], st[j - 1][i + (1 << (j - 1))]);
            }
        }
    }
    // Query interval [L, R] inclusive (0-indexed)
    T query(int L, int R) {
        int j = logs[R - L + 1];
        return combine(st[j][L], st[j][R - (1 << j) + 1]);
    }
};

//matrix multiplication:
vector<vector<long long>> multiply(vector<vector<long long>>& a, vector<vector<long long>>& b, long long mod) {
    int n = a.size();
    int m = a[0].size();
    int p = b[0].size();
    vector<vector<long long>> res(n, vector<long long>(p, 0));

    for (int i = 0; i < n; i++) {
        for (int k = 0; k < m; k++) {
            if (a[i][k] == 0) continue; 
            for (int j = 0; j < p; j++) {
                res[i][j] = (res[i][j] + a[i][k] * b[k][j]) % mod;
            }
        }
    }
    return res;
}

for(int i = 0; i < N; ++i) {
    for(int mask = 0; mask < (1 << N); mask++){
        if(mask & (1 << i)) {
            a[mask] += a[mask ^ (1 << i)];
        }
    }
}

//xor basis........................................................................................................
long long basis[62]; 
void insert(long long x) {
    for (int i = 61; i >= 0; i--) {
        if (!(x & (1LL << i))) continue;
        if (!basis[i]) {
            basis[i] = x;
            return;
        }
        x ^= basis[i];
    }
}

long long getMax() {
    long long res = 0;
    for (int i = 61; i >= 0; i--) {
        res = max(res, res ^ basis[i]);
    }
    return res;
}

const int MAXN = 1e5 + 5; // Total characters across all words
const int SIGMA = 26;    // Alphabet size

int trie[MAXN][SIGMA];
int cnt[MAXN];  // Number of words ending at this node
int pass[MAXN]; // Number of words passing through this node
int nodes = 1;  // Root is 1

void insert(string &s) {
    int curr = 1;
    for (char c : s) {
        int v = c - 'a';
        if (!trie[curr][v]) trie[curr][v] = ++nodes;
        curr = trie[curr][v];
        pass[curr]++; 
    }
    cnt[curr]++;
}

// "Update" by removing a word
void remove(string &s) {
    int curr = 1;
    // Optional: First check if s exists to avoid negative counts
    for (char c : s) {
        int v = c - 'a';
        curr = trie[curr][v];
        pass[curr]--;
    }
    cnt[curr]--;
}

bool search(string &s) {
    int curr = 1;
    for (char c : s) {
        int v = c - 'a';
        if (!trie[curr][v] || pass[trie[curr][v]] == 0) return false;
        curr = trie[curr][v];
    }
    return cnt[curr] > 0;
}

vector<int> dijkstra(int n, int start, vector<vector<pair<int,int>>>& adj) {
    vector<int> dist(n,INT_MAX);
    dist[start] = 0;
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    pq.push({0, start});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d>dist[u]) continue;
        for (auto& [v, w] : adj[u]) {
            if (dist[u]+w<dist[v]) {
                dist[v] = dist[u]+w;
                pq.push({dist[v],v});
            }
        }
    }
    return dist;
}

        auto [d, u] = pq.top(); pq.pop();
        if (d>dist[u]) continue;
        for (auto& [v, w] : adj[u]) {
            if (dist[u]+w<dist[v]) {
                dist[v] = dist[u]+w;
                pq.push({dist[v],v});
            }
        }
    }
    return dist;
}

//LCA………………………………………………………….

int up[MAXN][20], depth[MAXN];
void dfs_lca(int u, int p, int d, vector<vector<int>>& adj) {
    depth[u] = d;
    up[u][0] = p;
    for (int i = 1; i < 20; i++) 
        up[u][i] = up[up[u][i-1]][i-1];
    for (int v : adj[u]) 
        if (v != p) dfs_lca(v, u, d + 1, adj);
}

int get_lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);
    for (int i = 19; i >= 0; i--) 
        if (depth[u] - (1 << i) >= depth[v]) u = up[u][i];
    if (u == v) return u;
    for (int i = 19; i >= 0; i--) 
        if (up[u][i] != up[v][i]) { u = up[u][i]; v = up[v][i]; }
    return up[u][0];
}
